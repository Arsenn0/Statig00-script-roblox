local function onMenuButtonTouched(input)

       input.UserInputType == Enum.UserInputType.Touch then
        MenuVisible = not MenuVisible
        
        if MenuVisible then
            MainFrame.Size = UDim2.new(0, 0, 0, 0)
            MainFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
            MainFrame.Visible = true
            
            TweenService:Create(MainFrame, TweenInfo.new(0.3, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
                Size = UDim2.new(0, 340, 0, 460),
                Position = UDim2.new(0.5, -170, 0.5, -230)
            }):Play()
            
            TweenService:Create(IconContainer.Line1, TweenInfo.new(0.2), {
                Rotation = 45,
                Position = UDim2.new(0, 0, 0, 10)
            }):Play()
            TweenService:Create(IconContainer.Line2, TweenInfo.new(0.1), {
                BackgroundTransparency = 1
            }):Play()
            TweenService:Create(IconContainer.Line3, TweenInfo.new(0.2), {
                Rotation = -45,
                Position = UDim2.new(0, 0, 0, 10)
            }):Play()
        else
            TweenService:Create(MainFrame, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {
                Size = UDim2.new(0, 0, 0, 0),
                Position = UDim2.new(0.5, 0, 0.5, 0)
            }):Play()
            
            wait(0.2)
            MainFrame.Visible = false
            
            TweenService:Create(IconContainer.Line1, TweenInfo.new(0.2), {
                Rotation = 0,
                Position = UDim2.new(0, 0, 0, 0)
            }):Play()
            TweenService:Create(IconContainer.Line2, TweenInfo.new(0.1), {
                BackgroundTransparency = 0
            }):Play()
            TweenService:Create(IconContainer.Line3, TweenInfo.new(0.2), {
                Rotation = 0,
                Position = UDim2.new(0, 0, 0, 18)
            }):Play()
        end
    end
end

function CreateButton(parent, text, callback)
    local Button = Instance.new("TextButton")
    Button.Parent = parent
    Button.Size = UDim2.new(1, -10, 0, 36)
    Button.BackgroundColor3 = MediumPurple
    Button.Text = text
    Button.TextColor3 = WhiteColor
    Button.TextSize = 12
    Button.Font = Enum.Font.GothamMedium
    Button.BorderSizePixel = 0
    Button.ZIndex = 6
    Button.AutoButtonColor = false
    
    local BCorner = Instance.new("UICorner")
    BCorner.Parent = Button
    BCorner.CornerRadius = UDim.new(0, 8)
    
    local BStroke = Instance.new("UIStroke")
    BStroke.Parent = Button
    BStroke.Color = NeonPurple
    BStroke.Thickness = 1
    BStroke.Transparency = 0.5
    BStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    
    Button.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or 
           input.UserInputType == Enum.UserInputType.Touch then
            TweenService:Create(Button, TweenInfo.new(0.08), {
                BackgroundColor3 = NeonPurple
            }):Play()
            wait(0.08)
            TweenService:Create(Button, TweenInfo.new(0.08), {
                BackgroundColor3 = MediumPurple
            }):Play()
            callback()
        end
    end)
    
    return Button
end

function CreateSlider(parent, text, min, max, default, callback)
    local Frame = Instance.new("Frame")
    Frame.Parent = parent
    Frame.Size = UDim2.new(1, -10, 0, 55)
    Frame.BackgroundTransparency = 1

    Label.TextColor3 = LightPurple
    Label.TextSize = 11
    Label.Font = Enum.Font.GothamMedium

    Label.ZIndex = 6

    SliderFrame.Active = true
    
    local SliderCorner = Instance.new("UICorner")
    SliderCorner.Parent = SliderFrame
    SliderCorner.CornerRadius = UDim.new(1, 0)
    
    local SliderFill = Instance.new("Frame")
    SliderFill.Parent = SliderFrame
    SliderFill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
    SliderFill.BackgroundColor3 = NeonPurple
    SliderFill.ZIndex = 6
    
    local FillCorner = Instance.new("UICorner")
    FillCorner.Parent = SliderFill
    FillCorner.CornerRadius = UDim.new(1, 0)
    
    local SliderButton = Instance.new("TextButton")
    SliderButton.Parent = SliderFrame
    SliderButton.Size = UDim2.new(0, 16, 0, 16)
    SliderButton.Position = UDim2.new((default - min) / (max - min), -8, 0, -5)
    SliderButton.BackgroundColor3 = NeonPurple
    SliderButton.Text = ""
    SliderButton.BorderSizePixel = 0
    SliderButton.ZIndex = 7
    
    local BtnCorner = Instance.new("UICorner")
    BtnCorner.Parent = SliderButton
    BtnCorner.CornerRadius = UDim.new(1, 0)
    
    local BtnStroke = Instance.new("UIStroke")
    BtnStroke.Parent = SliderButton
    BtnStroke.Color = LightPurple
    BtnStroke.Thickness = 1.5
    BtnStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
    
    local dragging = false
    
    SliderButton.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or 
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
        end
    end)
    
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or 
           input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or 
                        input.UserInputType == Enum.UserInputType.Touch) then
            local mousePos = UserInputService:GetMouseLocation()
            local sliderPos = SliderFrame.AbsolutePosition.X
            local sliderWidth = SliderFrame.AbsoluteSize.X
            local percent = math.clamp((mousePos.X - sliderPos) / sliderWidth, 0, 1)
            local value = min + (max - min) * percent
            value = math.floor(value)
            SliderFill.Size = UDim2.new(percent, 0, 1, 0)
            SliderButton.Position = UDim2.new(percent, -8, 0, -5)
            Label.Text = text .. ": " .. value
            callback(value)
        end
    end)
    
    return Frame
end

-- Функция переключения вкладок
local function SwitchTab(tabIndex, tabName)
    for _, elem in pairs(MovementElements) do elem.Visible = false end
    for _, elem in pairs(VisualElements) do elem.Visible = false end
    for _, elem in pairs(TeleportElements) do elem.Visible = false end
    for _, elem in pairs(ServerElements) do elem.Visible = false end
    
    if tabIndex == 1 then
        for _, elem in pairs(MovementElements) do elem.Visible = true end
    elseif tabIndex == 2 then
        for _, elem in pairs(VisualElements) do elem.Visible = true end
    elseif tabIndex == 3 then

-- ИСПРАВЛЕННЫЙ FLY С ПОДДЕРЖКОЙ ВЗЛЁТА
-- ============================================
function StartFly()
    if FlyConnection then
        FlyConnection:Disconnect()
        FlyConnection = nil
    end
    
    local char = Player.Character or Player.CharacterAdded:Wait()
    if not char then return end
    
    local humanoid = char:WaitForChild("Humanoid")
    local rootPart = char:WaitForChild("HumanoidRootPart")
    
    -- Удаляем старые если есть
    if FlyBodyGyro then FlyBodyGyro:Destroy() end
    if FlyBodyVelocity then FlyBodyVelocity:Destroy() end
    
    -- Создаём BodyGyro для контроля направления
    FlyBodyGyro = Instance.new("BodyGyro")
    FlyBodyGyro.Parent = rootPart
    FlyBodyGyro.MaxTorque = Vector3.new(400000, 400000, 400000)
    FlyBodyGyro.P = 3000
    FlyBodyGyro.D = 500
    FlyBodyGyro.CFrame = rootPart.CFrame
    
    -- Создаём BodyVelocity для движения
    FlyBodyVelocity = Instance.new("BodyVelocity")
    FlyBodyVelocity.Parent = rootPart
    FlyBodyVelocity.MaxForce = Vector3.new(400000, 400000, 400000)
    FlyBodyVelocity.P = 3000
    FlyBodyVelocity.Velocity = Vector3.new(0, 0, 0)
    
    -- Основной цикл Fly через RenderStepped
    FlyConnection = RunService.RenderStepped:Connect(function()
        if not FlyEnabled or not char or not char.Parent or not rootPart or not rootPart.Parent then
            StopFly()
            return
        end
        
        -- Включаем PlatformStand чтобы отключить стандартную физику
        humanoid.PlatformStand = true
        
        -- Направление камеры
        local cameraCFrame = Camera.CFrame
        FlyBodyGyro.CFrame = CFrame.new(rootPart.Position, rootPart.Position + cameraCFrame.LookVector)
        
        -- Вектор движения
        local moveVector = Vector3.zero
        local flying = false
        
        -- Клавиатурное управление (для ПК и клавиатур)
        if UserInputService:IsKeyDown(Enum.KeyCode.W) then
            moveVector += cameraCFrame.LookVector
            flying = true
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.S) then
            moveVector -= cameraCFrame.LookVector
            flying = true
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then
            moveVector -= cameraCFrame.RightVector
            flying = true
        end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then
            moveVector += cameraCFrame.RightVector
            flying = true
        end
        
        -- ВВЕРХ - ПРОБЕЛ (самое важное исправление!)
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
            moveVector += Vector3.new(0, 1, 0)
            flying = true
        end
        
        -- ВНИЗ - LeftShift или LeftControl
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) or UserInputService:IsKeyDown(Enum.KeyCode.LeftControl) then
            moveVector -= Vector3.new(0, 1, 0)
            flying = true
        end
        
        -- Мобильное управление через MoveDirection (джойстик)
        if not flying and humanoid.MoveDirection.Magnitude > 0 then
            local moveDir = humanoid.MoveDirection
            -- Конвертируем 2D направление в 3D относительно камеры
            local cameraForward = Vector3.new(cameraCFrame.LookVector.X, 0, cameraCFrame.LookVector.Z).Unit
            local cameraRight = Vector3.new(cameraCFrame.RightVector.X, 0, cameraCFrame.RightVector.Z).Unit
            
            moveVector = cameraForward * moveDir.Z + cameraRight * moveDir.X
            
            -- Если смотрит вверх - летим вверх по направлению камеры
            if cameraCFrame.LookVector.Y > 0.3 then
                moveVector += Vector3.new(0, cameraCFrame.LookVector.Y, 0)
            end
            -- Если смотрит вниз - летим вниз
            if cameraCFrame.LookVector.Y < -0.3 then
                moveVector += Vector3.new(0, cameraCFrame.LookVector.Y, 0)
            end
            
            flying = true
        end
        
        -- Применяем скорость
        if flying then
            FlyBodyVelocity.Velocity = moveVector.Unit * FlySpeed
        else
            -- Зависаем на месте
            FlyBodyVelocity.Velocity = Vector3.new(0, 0, 0)
        end
    end)
end

function StopFly()
    if FlyConnection then
        FlyConnection:Disconnect()
        FlyConnection = nil
    end
    if FlyBodyGyro then
        FlyBodyGyro:Destroy()
        FlyBodyGyro = nil
    end
    if FlyBodyVelocity then
        FlyBodyVelocity:Destroy()
        FlyBodyVelocity = nil
    end
    if Player.Character and Player.Character:FindFirstChild("Humanoid") then
        Player.Character.Humanoid.PlatformStand = false
    end
end

-- ПРИТЯГИВАНИЕ ИГРОКОВ
-- ============================================
function StartAttract()
    spawn(function()
        while AttractPlayersEnabled do
            if Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") then
                local root = Player.Character.HumanoidRootPart
                for _, plr in pairs(game:GetService("Players"):GetPlayers()) do
                    if plr ~= Player and plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                        local targetRoot = plr.Character.HumanoidRootPart
                        local distance = (targetRoot.Position - root.Position).Magnitude
                        if distance <= AttractRadius and distance > 3 then
                            local direction = (root.Position - targetRoot.Position).Unit
                            local force = math.min((AttractRadius - distance) / AttractRadius, 1) * 2
                            targetRoot.Velocity = direction * force * 25
                            targetRoot.CFrame = targetRoot.CFrame + direction * force * 0.5
                        end
                    end
                end
            end
            wait(0.05)
        end
    end)
end

-- ESP
-- ============================================
local ESPTable = {}

function EnableESP()
    local function CreateESP(plr)
        if plr == Player then return end
        if ESPTable[plr] then return end
        
        local highlight = Instance.new("Highlight")
        highlight.Name = "ESP_" .. plr.Name
        highlight.FillColor = NeonPurple
        highlight.FillTransparency = 0.5
        highlight.OutlineColor = LightPurple
        highlight.OutlineTransparency = 0
        ESPTable[plr] = highlight
        
        if plr.Character then
            highlight.Parent = plr.Character
        end
        
        plr.CharacterAdded:Connect(function(char)
            wait(0.1)
            highlight.Parent = char
        end)
    end
    
    for _, plr in pairs(game:GetService("Players"):GetPlayers()) do
        if plr ~= Player then
            CreateESP(plr)
        end
    end
    
    game:GetService("Players").PlayerAdded:Connect(function(plr)
        CreateESP(plr)
    end)
    
    game:GetService("Players").PlayerRemoving:Connect(function(plr)
        if ESPTable[plr] then
            ESPTable[plr]:Destroy()
            ESPTable[plr] = nil
        end
    end)
end

function DisableESP()
    for plr, highlight in pairs(ESPTable) do
        highlight:Destroy()
    end
    ESPTable = {}
end
